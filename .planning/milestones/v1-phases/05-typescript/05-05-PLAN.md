```typescript
import type { CoherentNode, CoherentElement, StrictCoherentElement } from '@coherent.js/core';

// Typed form field definition
export interface FormField<T = unknown> {
  name: string;
  type: 'text' | 'email' | 'password' | 'number' | 'checkbox' | 'radio' | 'select' | 'textarea' | 'file' | 'date' | 'hidden';
  label?: string;
  placeholder?: string;
  required?: boolean;
  disabled?: boolean;
  readonly?: boolean;
  defaultValue?: T;
  validation?: FieldValidation<T>;
  options?: SelectOption[];  // For select/radio
  transform?: (value: unknown) => T;
}

export interface FieldValidation<T = unknown> {
  required?: boolean | string;  // String is custom message
  minLength?: number | { value: number; message: string };
  maxLength?: number | { value: number; message: string };
  min?: number | { value: number; message: string };
  max?: number | { value: number; message: string };
  pattern?: RegExp | { value: RegExp; message: string };
  custom?: (value: T, formData: Record<string, unknown>) => boolean | string | Promise<boolean | string>;
}

export interface SelectOption {
  value: string | number;
  label: string;
  disabled?: boolean;
}

// Typed form builder
export interface FormBuilder<T extends Record<string, unknown>> {
  addField<K extends keyof T>(name: K, field: Omit<FormField<T[K]>, 'name'>): FormBuilder<T>;
  setAction(action: string): FormBuilder<T>;
  setMethod(method: 'get' | 'post'): FormBuilder<T>;
  onSubmit(handler: (data: T) => void | Promise<void>): FormBuilder<T>;
  build(): CoherentNode;
  validate(data: unknown): { valid: boolean; errors: Record<keyof T, string> };
}

export function createForm<T extends Record<string, unknown>>(
  config?: FormConfig
): FormBuilder<T>;

export function renderField(field: FormField): CoherentNode;
export function validateField<T>(field: FormField<T>, value: unknown): string | null;
```

```typescript
import type { CoherentNode } from '@coherent.js/core';

// Translation key type for type-safe translations
export type TranslationKey = string;

// Nested translation object type
export type TranslationMessages = {
  [key: string]: string | TranslationMessages;
};

// Translation function with interpolation
export type TranslationFunction = {
  (key: TranslationKey): string;
  (key: TranslationKey, params: Record<string, string | number>): string;
  (key: TranslationKey, params: Record<string, string | number>, count: number): string;
};

export interface I18nConfig {
  defaultLocale: string;
  supportedLocales: string[];
  fallbackLocale?: string;
  loadMessages?: (locale: string) => Promise<TranslationMessages>;
  messages?: Record<string, TranslationMessages>;
  missingKeyHandler?: (key: string, locale: string) => string;
}

export interface I18nInstance {
  locale: string;
  t: TranslationFunction;
  setLocale(locale: string): Promise<void>;
  getLocale(): string;
  getSupportedLocales(): string[];
  hasTranslation(key: TranslationKey): boolean;
  formatNumber(value: number, options?: Intl.NumberFormatOptions): string;
  formatDate(value: Date, options?: Intl.DateTimeFormatOptions): string;
  formatCurrency(value: number, currency: string, options?: Intl.NumberFormatOptions): string;
}

// Typed translation helper for known keys
export function createTypedTranslator<T extends TranslationMessages>(
  messages: T
): (key: keyof FlattenKeys<T>, params?: Record<string, string | number>) => string;

// Helper type to flatten nested keys
type FlattenKeys<T, Prefix extends string = ''> = T extends string
  ? Prefix
  : { [K in keyof T]: FlattenKeys<T[K], `${Prefix}${Prefix extends '' ? '' : '.'}${K & string}`> }[keyof T];

export function createI18n(config: I18nConfig): I18nInstance;
export function useTranslation(): TranslationFunction;
export function useLocale(): [string, (locale: string) => Promise<void>];
```

```typescript
// Typed state store
export interface Store<T extends Record<string, unknown>> {
  getState(): T;
  setState(partial: Partial<T> | ((state: T) => Partial<T>)): void;
  subscribe(listener: (state: T, prevState: T) => void): () => void;
  destroy(): void;
}

export interface StoreOptions<T> {
  initialState: T;
  persist?: {
    key: string;
    storage?: Storage;
    serialize?: (state: T) => string;
    deserialize?: (value: string) => T;
  };
  devtools?: boolean;
  name?: string;
}

export function createStore<T extends Record<string, unknown>>(
  options: StoreOptions<T>
): Store<T>;

// Selector support
export function createSelector<T, R>(
  store: Store<T>,
  selector: (state: T) => R
): () => R;

// Action helpers
export type Action<T, P = void> = P extends void
  ? () => Partial<T>
  : (payload: P) => Partial<T>;

export function createActions<T, A extends Record<string, Action<T, any>>>(
  actions: A
): { [K in keyof A]: A[K] extends Action<T, infer P> ? (payload: P) => void : never };
```
