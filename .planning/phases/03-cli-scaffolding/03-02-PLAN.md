/**
 * Extract import statements from code
 */
function extractImports(code) {
  const imports = [];

  // Match: import { a, b } from 'pkg'
  const namedRegex = /import\s*\{([^}]+)\}\s*from\s*['"]([^'"]+)['"]/g;
  let match;
  while ((match = namedRegex.exec(code)) !== null) {
    const names = match[1].split(',').map(s => s.trim().split(/\s+as\s+/)[0].trim());
    imports.push({ names, from: match[2] });
  }

  // Match: import x from 'pkg'
  const defaultRegex = /import\s+(\w+)\s+from\s*['"]([^'"]+)['"]/g;
  while ((match = defaultRegex.exec(code)) !== null) {
    imports.push({ names: [match[1]], from: match[2], isDefault: true });
  }

  return imports;
}

/**
 * Validate imports against known exports
 */
function validateImports(imports) {
  const errors = [];

  for (const imp of imports) {
    if (imp.from.startsWith('@coherent.js/')) {
      const pkgExports = PACKAGE_EXPORTS[imp.from];

      if (!pkgExports) {
        errors.push(`Unknown package: ${imp.from}`);
        continue;
      }

      for (const name of imp.names) {
        if (!pkgExports.includes(name) && !imp.isDefault) {
          errors.push(`${imp.from} does not export '${name}'`);
        }
      }
    }
  }

  return errors;
}

/**
 * Recursively get all JS/TS files in directory
 */
async function getAllSourceFiles(dir, files = []) {
  const entries = await readdir(dir, { withFileTypes: true });

  for (const entry of entries) {
    const fullPath = join(dir, entry.name);
    if (entry.isDirectory() && entry.name !== 'node_modules') {
      await getAllSourceFiles(fullPath, files);
    } else if (entry.isFile() && /\.(js|ts)$/.test(entry.name)) {
      files.push(fullPath);
    }
  }

  return files;
}
