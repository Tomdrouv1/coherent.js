/**
 * Clean hydrate() API for Coherent.js
 *
 * Integrates event delegation, state serialization, and mismatch detection
 * into a simple function that returns a control object.
 */

import { eventDelegation, handlerRegistry } from './events/index.js';
import { extractState, serializeState } from './hydration/index.js';
import { detectMismatch, reportMismatches } from './hydration/index.js';

// Track hydrated instances by container
const hydratedInstances = new WeakMap();

// Detect development mode
const isDevelopment = typeof process !== 'undefined' &&
  process.env?.NODE_ENV !== 'production';

/**
 * Hydrate a server-rendered component
 *
 * @param {Function} component - Component function
 * @param {Element} container - DOM container element
 * @param {Object} options - Hydration options
 * @param {Object} options.props - Props to pass to component
 * @param {Object} options.initialState - Initial state (overrides extracted state)
 * @param {boolean} options.strict - Throw on mismatch instead of warn
 * @param {Function} options.onMismatch - Custom mismatch handler
 * @param {boolean} options.detectMismatch - Enable mismatch detection (default: dev only)
 * @returns {Object|null} Control object with unmount() and rerender()
 */
export function hydrate(component, container, options = {}) {
  // Validate inputs
  if (typeof component !== 'function') {
    console.error('[Coherent.js] hydrate(): component must be a function');
    return null;
  }

  if (!container || typeof container.getAttribute !== 'function') {
    console.error('[Coherent.js] hydrate(): container must be a DOM element');
    return null;
  }

  // Check if already hydrated
  if (hydratedInstances.has(container)) {
    console.warn('[Coherent.js] hydrate(): container already hydrated');
    return hydratedInstances.get(container);
  }

  const config = {
    props: {},
    initialState: null,
    strict: false,
    onMismatch: null,
    detectMismatch: isDevelopment,
    ...options
  };

  // Initialize event delegation (idempotent)
  eventDelegation.initialize();

  // Extract state from container
  const extractedState = extractState(container);
  const state = config.initialState || extractedState || {};

  // Generate virtual DOM from component
  const props = { ...config.props, ...state };
  let virtualDOM;
  try {
    virtualDOM = component(props);
  } catch (err) {
    console.error('[Coherent.js] hydrate(): component render error', err);
    return null;
  }

  // Detect mismatches if enabled
  if (config.detectMismatch && virtualDOM) {
    const mismatches = detectMismatch(container, virtualDOM);

    if (mismatches.length > 0) {
      const componentName = component.name || 'Unknown';

      if (config.onMismatch) {
        config.onMismatch(mismatches);
      } else {
        reportMismatches(mismatches, {
          componentName,
          strict: config.strict
        });
      }

      // In strict mode, reportMismatches throws, so we won't reach here
    }
  }

  // Create instance
  const instance = createInstance(container, component, props, state);

  // Register event handlers from virtual DOM
  registerEventHandlers(container, virtualDOM, instance);

  // Store instance
  hydratedInstances.set(container, instance.control);
  container.__coherentInstance = instance;

  return instance.control;
}

/**
 * Create a component instance with state management
 * @private
 */
function createInstance(container, component, props, initialState) {
  const handlerIds = new Set();
  let currentState = { ...initialState };
  let currentProps = { ...props };

  const instance = {
    container,
    component,
    handlerIds,

    get state() {
      return currentState;
    },

    get props() {
      return currentProps;
    },

    setState(newState) {
      const prevState = currentState;
      currentState = typeof newState === 'function'
        ? { ...currentState, ...newState(currentState) }
        : { ...currentState, ...newState };

      // Trigger re-render
      this.rerender();

      // Optional: callback
      if (instance.onStateChange) {
        instance.onStateChange(currentState, prevState);
      }
    },

    setProps(newProps) {
      currentProps = { ...currentProps, ...newProps };
      this.rerender();
    },

    rerender() {
      try {
        const mergedProps = { ...currentProps, ...currentState };
        const newVirtualDOM = component(mergedProps);

        // Patch DOM using existing patchDOM logic
        patchDOM(container, newVirtualDOM, instance);

        // Re-register event handlers
        registerEventHandlers(container, newVirtualDOM, instance);

      } catch (err) {
        console.error('[Coherent.js] rerender error:', err);
      }
    },

    control: {
      unmount() {
        // Clean up handlers
        handlerIds.forEach(id => handlerRegistry.unregister(id));
        handlerIds.clear();

        // Remove instance
        hydratedInstances.delete(container);
        delete container.__coherentInstance;
      },

      rerender() {
        instance.rerender();
      },

      getState() {
        return currentState;
      },

      setState(newState) {
        instance.setState(newState);
      }
    }
  };

  return instance;
}

/**
 * Register event handlers from virtual DOM
 * @private
 */
function registerEventHandlers(element, virtualNode, instance, path = '') {
  if (!virtualNode || typeof virtualNode !== 'object') return;

  if (Array.isArray(virtualNode)) {
    virtualNode.forEach((child, i) => {
      const childElement = element.children[i];
      if (childElement) {
        registerEventHandlers(childElement, child, instance, `${path}[${i}]`);
      }
    });
    return;
  }

  const tagName = Object.keys(virtualNode)[0];
  const props = virtualNode[tagName] || {};

  // Find event handlers (onclick, onchange, etc.)
  const eventProps = Object.keys(props).filter(key =>
    key.startsWith('on') && typeof props[key] === 'function'
  );

  eventProps.forEach(eventProp => {
    const eventType = eventProp.substring(2).toLowerCase(); // onclick -> click
    const handler = props[eventProp];

    // Generate unique handler ID
    const handlerId = generateHandlerId(instance.component.name, eventType, path);

    // Set data attribute on element
    element.setAttribute(`data-coherent-${eventType}`, handlerId);

    // Register handler
    handlerRegistry.register(handlerId, handler, {
      component: instance.component,
      state: instance.state,
      setState: instance.setState.bind(instance),
      props: instance.props
    });

    // Track for cleanup
    instance.handlerIds.add(handlerId);
  });

  // Process children
  if (props.children) {
    const children = Array.isArray(props.children) ? props.children : [props.children];
    children.forEach((child, i) => {
      const childElement = element.children[i];
      if (childElement && typeof child === 'object') {
        registerEventHandlers(childElement, child, instance, `${path}.children[${i}]`);
      }
    });
  }
}

/**
 * Generate unique handler ID
 * @private
 */
let handlerCounter = 0;
function generateHandlerId(componentName, eventType, path) {
  return `${componentName || 'c'}_${eventType}_${++handlerCounter}`;
}

/**
 * Patch DOM with new virtual DOM
 * Uses key-based reconciliation from Phase 1
 * @private
 */
function patchDOM(container, virtualNode, instance) {
  // Delegate to existing patching logic
  // This function should use the patchChildren/patchAttributes from hydration.js
  // For now, implement basic patching

  if (!virtualNode || typeof virtualNode !== 'object') return;

  const tagName = Object.keys(virtualNode)[0];
  const props = virtualNode[tagName] || {};

  // Update attributes
  patchAttributes(container, props);

  // Update children
  if (props.children !== undefined || props.text !== undefined) {
    patchChildren(container, props, instance);
  }
}

/**
 * Patch element attributes
 * @private
 */
function patchAttributes(element, props) {
  // Update className
  if (props.className !== undefined) {
    element.setAttribute('class', props.className);
  }

  // Update other attributes (skip children, text, event handlers, key)
  Object.keys(props).forEach(key => {
    if (key === 'children' || key === 'text' || key === 'key') return;
    if (key.startsWith('on')) return; // Event handlers handled separately
    if (key === 'className') return; // Already handled

    const value = props[key];
    const attrName = key;

    if (value === true) {
      element.setAttribute(attrName, '');
    } else if (value === false || value === null || value === undefined) {
      element.removeAttribute(attrName);
    } else {
      element.setAttribute(attrName, String(value));
    }
  });
}

/**
 * Patch children with key-based reconciliation
 * @private
 */
function patchChildren(container, props, instance) {
  // Get new children
  let newChildren = [];
  if (props.children) {
    newChildren = Array.isArray(props.children) ? props.children : [props.children];
  } else if (props.text !== undefined) {
    newChildren = [String(props.text)];
  }

  // Get current DOM children (significant only)
  const currentChildren = Array.from(container.childNodes).filter(node =>
    node.nodeType === 1 || (node.nodeType === 3 && node.textContent?.trim())
  );

  // Build key maps for reconciliation
  const oldKeyMap = new Map();
  const newKeyMap = new Map();

  currentChildren.forEach((child, i) => {
    const key = child.getAttribute?.('data-key') || `__index_${i}`;
    oldKeyMap.set(key, { node: child, index: i });
  });

  newChildren.forEach((child, i) => {
    const key = getVNodeKey(child) || `__index_${i}`;
    newKeyMap.set(key, { vnode: child, index: i });
  });

  // Process new children
  const usedOldKeys = new Set();

  newChildren.forEach((vChild, newIndex) => {
    const key = getVNodeKey(vChild) || `__index_${newIndex}`;
    const oldEntry = oldKeyMap.get(key);

    if (oldEntry) {
      // Reuse existing node
      usedOldKeys.add(key);
      patchNode(oldEntry.node, vChild, instance);

      // Move if needed
      if (newIndex < container.children.length) {
        const referenceNode = container.childNodes[newIndex];
        if (referenceNode !== oldEntry.node) {
          container.insertBefore(oldEntry.node, referenceNode);
        }
      } else {
        container.appendChild(oldEntry.node);
      }
    } else {
      // Create new node
      const newNode = createNode(vChild, instance);
      if (newIndex < container.children.length) {
        container.insertBefore(newNode, container.childNodes[newIndex]);
      } else {
        container.appendChild(newNode);
      }
    }
  });

  // Remove unused old nodes
  oldKeyMap.forEach((entry, key) => {
    if (!usedOldKeys.has(key)) {
      entry.node.remove();
    }
  });
}

/**
 * Get key from virtual node
 * @private
 */
function getVNodeKey(vNode) {
  if (!vNode || typeof vNode !== 'object' || Array.isArray(vNode)) return undefined;
  const tagName = Object.keys(vNode)[0];
  const props = vNode[tagName];
  return props?.key;
}

/**
 * Patch a single node
 * @private
 */
function patchNode(domNode, vNode, instance) {
  if (typeof vNode === 'string' || typeof vNode === 'number') {
    if (domNode.nodeType === 3) {
      if (domNode.textContent !== String(vNode)) {
        domNode.textContent = String(vNode);
      }
    }
    return;
  }

  if (!vNode || typeof vNode !== 'object') return;

  const tagName = Object.keys(vNode)[0];
  const props = vNode[tagName] || {};

  // Check tag match
  if (domNode.tagName?.toLowerCase() !== tagName.toLowerCase()) {
    // Replace node
    const newNode = createNode(vNode, instance);
    domNode.parentNode?.replaceChild(newNode, domNode);
    return;
  }

  patchAttributes(domNode, props);

  if (props.children !== undefined || props.text !== undefined) {
    patchChildren(domNode, props, instance);
  }
}

/**
 * Create DOM node from virtual node
 * @private
 */
function createNode(vNode, instance) {
  if (typeof vNode === 'string' || typeof vNode === 'number') {
    return document.createTextNode(String(vNode));
  }

  if (!vNode || typeof vNode !== 'object') {
    return document.createTextNode('');
  }

  if (Array.isArray(vNode)) {
    const fragment = document.createDocumentFragment();
    vNode.forEach(child => fragment.appendChild(createNode(child, instance)));
    return fragment;
  }

  const tagName = Object.keys(vNode)[0];
  const props = vNode[tagName] || {};
  const element = document.createElement(tagName);

  // Set attributes
  Object.keys(props).forEach(key => {
    if (key === 'children' || key === 'text' || key === 'key') return;
    if (key.startsWith('on')) return;

    const value = props[key];
    const attrName = key === 'className' ? 'class' : key;

    if (value === true) {
      element.setAttribute(attrName, '');
    } else if (value !== false && value !== null && value !== undefined) {
      element.setAttribute(attrName, String(value));
    }
  });

  // Set key as data attribute for reconciliation
  if (props.key !== undefined) {
    element.setAttribute('data-key', props.key);
  }

  // Add children
  if (props.children) {
    const children = Array.isArray(props.children) ? props.children : [props.children];
    children.forEach(child => element.appendChild(createNode(child, instance)));
  } else if (props.text !== undefined) {
    element.appendChild(document.createTextNode(String(props.text)));
  }

  // Register event handlers
  registerEventHandlers(element, vNode, instance, '');

  return element;
}

export default hydrate;
