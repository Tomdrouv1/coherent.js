  <files>packages/client/types/index.d.ts</files>
  <action>
Update index.d.ts to import and use types from @coherent.js/core where appropriate:

1. Add import at the top of the file:
```typescript
import type {
  CoherentNode,
  CoherentElement,
  StrictCoherentElement,
  CoherentChild,
  CoherentComponent,
  ComponentProps
} from '@coherent.js/core';
```

2. Update the local CoherentElement interface to extend or reference core's:
- Keep local CoherentElement for HTMLElement extensions (DOM-specific properties)
- The existing `CoherentElement extends HTMLElement` is correct for client-side DOM elements
- Rename it to `CoherentHTMLElement` to avoid confusion with core's CoherentElement

3. Update function signatures that accept components to use core types:
```typescript
export function hydrate(
  element: HTMLElement,
  component: CoherentComponent | ((props?: any) => CoherentNode),
  props?: Record<string, any>,
  options?: { initialState?: any }
): HydratedInstance | null;
```

4. Update HydratedInstance to reference core types:
```typescript
export interface HydratedInstance {
  element: HTMLElement;
  component: CoherentComponent;
  props: ComponentProps;
  state: any;
  isHydrated: boolean;
  // ... methods
}
```

5. Update event handler types to be more specific (match DOM events):
```typescript
export type EventHandler<E extends Event = Event> = (
  event: E,
  element: HTMLElement,
  data?: any
) => void | Promise<void>;

// Specific handlers
export type ClickHandler = EventHandler<MouseEvent>;
export type KeyHandler = EventHandler<KeyboardEvent>;
export type FocusHandler = EventHandler<FocusEvent>;
export type SubmitHandler = EventHandler<SubmitEvent>;
export type ChangeHandler = EventHandler<Event>;
```

6. Ensure SerializableState matches what can actually be serialized:
```typescript
export type SerializableValue =
  | string
  | number
  | boolean
  | null
  | SerializableValue[]
  | { [key: string]: SerializableValue };

export interface SerializableState {
  [key: string]: SerializableValue;
}
```

Keep backward compatibility - don't remove existing types, add new more specific ones as alternatives.

import { expectTypeOf } from 'vitest';
import {
  hydrate,
  hydrateAll,
  hydrateBySelector,
  autoHydrate,
  enableClientEvents,
  makeHydratable,
  registerEventHandler
} from '@coherent.js/client';

import type {
  HydratedInstance,
  HydrationOptions,
  SerializableState,
  EventHandler,
  ClientComponent,
  BatchHydrationResult
} from '@coherent.js/client';

import type {
  CoherentComponent,
  CoherentNode
} from '@coherent.js/core';

// ============================================================================
// hydrate() Function
// ============================================================================

// hydrate returns HydratedInstance or null
expectTypeOf(hydrate).returns.toMatchTypeOf<HydratedInstance | null>();

// hydrate accepts HTMLElement as first parameter
expectTypeOf(hydrate).parameter(0).toMatchTypeOf<HTMLElement>();

// hydrate accepts component function
expectTypeOf(hydrate).parameter(1).toMatchTypeOf<Function>();

// hydrate accepts optional props
expectTypeOf(hydrate).parameter(2).toMatchTypeOf<Record<string, any> | undefined>();

// hydrate accepts optional options with initialState
expectTypeOf(hydrate).parameter(3).toMatchTypeOf<{ initialState?: any } | undefined>();

// HydratedInstance has expected properties
declare const instance: HydratedInstance;
expectTypeOf(instance.element).toMatchTypeOf<HTMLElement>();
expectTypeOf(instance.isHydrated).toBeBoolean();
expectTypeOf(instance.update).toBeFunction();
expectTypeOf(instance.rerender).toBeFunction();
expectTypeOf(instance.destroy).toBeFunction();
expectTypeOf(instance.setState).toBeFunction();

// ============================================================================
// hydrateAll() Function
// ============================================================================

// hydrateAll returns array of HydratedInstance | null
expectTypeOf(hydrateAll).returns.toMatchTypeOf<Array<HydratedInstance | null>>();

// hydrateAll accepts arrays
expectTypeOf(hydrateAll).parameter(0).toMatchTypeOf<HTMLElement[]>();
expectTypeOf(hydrateAll).parameter(1).toMatchTypeOf<Function[]>();

// ============================================================================
// hydrateBySelector() Function
// ============================================================================

// hydrateBySelector returns array of HydratedInstance | null
expectTypeOf(hydrateBySelector).returns.toMatchTypeOf<Array<HydratedInstance | null>>();

// hydrateBySelector accepts selector string
expectTypeOf(hydrateBySelector).parameter(0).toBeString();
expectTypeOf(hydrateBySelector).parameter(1).toMatchTypeOf<Function>();

// ============================================================================
// autoHydrate() Function
// ============================================================================

// autoHydrate returns void
expectTypeOf(autoHydrate).returns.toBeVoid();

// autoHydrate accepts optional component registry
expectTypeOf(autoHydrate).parameter(0).toMatchTypeOf<Record<string, any> | undefined>();

// ============================================================================
// enableClientEvents() Function
// ============================================================================

// enableClientEvents returns void
expectTypeOf(enableClientEvents).returns.toBeVoid();

// enableClientEvents accepts optional root element
expectTypeOf(enableClientEvents).parameter(0).toMatchTypeOf<Document | HTMLElement | undefined>();

// ============================================================================
// makeHydratable() Function
// ============================================================================

// makeHydratable returns same function type
expectTypeOf(makeHydratable).toBeFunction();

// Generic preservation
const MyComponent = (props: { name: string }) => ({ div: { text: props.name } });
const HydratableComponent = makeHydratable(MyComponent);
expectTypeOf(HydratableComponent).toMatchTypeOf<typeof MyComponent>();

// ============================================================================
// registerEventHandler() Function
// ============================================================================

// registerEventHandler returns void
expectTypeOf(registerEventHandler).returns.toBeVoid();

// registerEventHandler accepts id and handler
expectTypeOf(registerEventHandler).parameter(0).toBeString();
expectTypeOf(registerEventHandler).parameter(1).toMatchTypeOf<Function>();

// ============================================================================
// SerializableState Type
// ============================================================================

// Valid serializable states
const validState1: SerializableState = { count: 0 };
const validState2: SerializableState = { name: 'test', active: true };
const validState3: SerializableState = { items: ['a', 'b', 'c'] };
const validState4: SerializableState = { nested: { value: 42 } };
const validState5: SerializableState = { nullable: null };

expectTypeOf(validState1).toMatchTypeOf<SerializableState>();
expectTypeOf(validState2).toMatchTypeOf<SerializableState>();
expectTypeOf(validState3).toMatchTypeOf<SerializableState>();
expectTypeOf(validState4).toMatchTypeOf<SerializableState>();
expectTypeOf(validState5).toMatchTypeOf<SerializableState>();

// EventHandler accepts Event
const handler: EventHandler = (event, element, data) => {
  expectTypeOf(event).toMatchTypeOf<Event>();

// Handler can return void or Promise<void>

expectTypeOf(clientComponent.getState).returns.toMatchTypeOf<SerializableState>();
expectTypeOf(clientComponent.resetState).toBeFunction();

// Lifecycle
expectTypeOf(clientComponent.render).toBeFunction();
expectTypeOf(clientComponent.destroy).toBeFunction();
expectTypeOf(clientComponent.refresh).toBeFunction();

// Events
expectTypeOf(clientComponent.addEventListener).toBeFunction();
expectTypeOf(clientComponent.removeEventListener).toBeFunction();
expectTypeOf(clientComponent.trigger).toBeFunction();

// Serialization
expectTypeOf(clientComponent.serialize).returns.toBeString();
expectTypeOf(clientComponent.toJSON).returns.toMatchTypeOf<SerializableState>();

// Properties
expectTypeOf(clientComponent.element).toMatchTypeOf<HTMLElement>();
expectTypeOf(clientComponent.isHydrated).toBeBoolean();
expectTypeOf(clientComponent.id).toBeString();

// ============================================================================
// Integration with Core Types

// Component using core types works with hydrate
