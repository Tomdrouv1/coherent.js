1. State Management with withState

Advanced Form Component
export const ContactForm = withState({
  formData: { name: '', email: '', message: '' },
  errors: {},
  isSubmitting: false
})(({ state, setState, stateUtils }) => {
  
  const handleSubmit = async (event) => {
    event.preventDefault();
    const errors = validateForm(state.formData);
    
    if (Object.keys(errors).length > 0) {
      setState({ errors });
      return;
    }
    
    setState({ isSubmitting: true });
    
    try {
      await submitForm(state.formData);
      stateUtils.resetState();
    } catch (error) {
      setState({ isSubmitting: false, errors: { submit: error.message } });
    }
  };
  
  return {
    form: {
      onsubmit: handleSubmit,
      children: [
        // Form fields...
      ]
    }
  };
});

2. Performance-Optimized Components

Memoized Product List
export const ProductList = memo(
  ({ products, filters, sortBy }) => {
    const filteredProducts = products
      .filter(product => applyFilters(product, filters))
      .sort((a, b) => sortProducts(a, b, sortBy));
    
    return {
      div: {
        className: 'product-grid',
        children: filteredProducts.map(product => 
          ProductCard({ product, key: product.id })
        )
      }
    };
  },
  // Custom memoization key
  ({ products, filters, sortBy }) => 
    `${products.length}-${JSON.stringify(filters)}-${sortBy}`
);

3. Reusable Component Patterns

Button Component with Variants
export const Button = ({ 
  variant = 'primary', 
  size = 'md', 
  disabled = false, 
  onClick, 
  children, 
  ...props 
}) => ({
  button: {
    className: `btn btn-${variant} btn-${size} ${disabled ? 'btn-disabled' : ''}`,
    disabled,
    onclick: onClick,
    'aria-disabled': disabled,
    ...props,
    children: Array.isArray(children) ? children : [children]
  }
});

// Usage
Button({
  variant: 'success',

  ]
})
